-- Execute no SQL Editor do seu projeto

-- Tabela principal de peças
CREATE TABLE IF NOT EXISTS pecas (
  id            uuid          DEFAULT gen_random_uuid() PRIMARY KEY,
  marca         text          NOT NULL,
  modelo        text          NOT NULL,
  ano           integer       NOT NULL,
  motor         text          NOT NULL,
  codigo_oem    text          NOT NULL,
  descricao     text,
  preco_medio   numeric(10,2),
  foto_url      text,
  created_at    timestamptz   DEFAULT now(),
  updated_at    timestamptz   DEFAULT now()
);

-- Índices compostos para filtros em cascata (performance)
CREATE INDEX IF NOT EXISTS idx_marca   ON pecas(marca);
CREATE INDEX IF NOT EXISTS idx_modelo  ON pecas(marca, modelo);
CREATE INDEX IF NOT EXISTS idx_ano     ON pecas(marca, modelo, ano);
CREATE INDEX IF NOT EXISTS idx_motor   ON pecas(marca, modelo, ano, motor);
CREATE INDEX IF NOT EXISTS idx_oem     ON pecas(codigo_oem);

-- Previne duplicatas exatas
CREATE UNIQUE INDEX IF NOT EXISTS idx_unique_peca
  ON pecas(marca, modelo, ano, motor, codigo_oem);

-- Row Level Security
ALTER TABLE pecas ENABLE ROW LEVEL SECURITY;
CREATE POLICY "leitura_publica"
  ON pecas FOR SELECT USING (true);
CREATE POLICY "insercao_service"
  ON pecas FOR INSERT WITH CHECK (true);

-- Trigger para updated_at automático
CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER trg_updated_at
  BEFORE UPDATE ON pecas
  FOR EACH ROW EXECUTE FUNCTION set_updated_at();
